## 🎯 核心功能示例代码

### 1. 人群计算Spark作业

创建 `spark-jobs/crowd-compute.py`:

```python
from pyspark.sql import SparkSession
from pyspark.sql.functions import *
import json
import sys

def create_spark_session():
    return SparkSession.builder \
        .appName("CrowdCompute") \
        .config("hive.metastore.uris", "thrift://hive-metastore:9083") \
        .enableHiveSupport() \
        .getOrCreate()

def parse_crowd_rules(rules_json):
    """解析人群规则JSON"""
    rules = json.loads(rules_json)
    conditions = []
    
    for condition in rules['conditions']:
        tag = condition['tag'] 
        operator = condition['operator']
        values = condition['values']
        
        if operator == 'in':
            conditions.append(f"{tag} IN ({','.join([repr(v) for v in values])})")
        elif operator == '>=':
            conditions.append(f"{tag} >= '{values[0]}'")
        elif operator == '=':
            conditions.append(f"{tag} = '{values[0]}'")
    
    return f" {rules['operator']} ".join(conditions)

def compute_crowd(spark, crowd_id, rules_json, dt):
    """计算人群"""
    
    # 解析规则
    where_clause = parse_crowd_rules(rules_json)
    
    # 查询用户数据
    sql = f"""
    SELECT uid, '{crowd_id}' as crowd_code, '{dt}' as compute_dt
    FROM marketing_platform.user_tags 
    WHERE dt = '{dt}' AND ({where_clause})
    """
    
    print(f"执行SQL: {sql}")
    result_df = spark.sql(sql)
    
    # 显示结果统计
    count = result_df.count()
    print(f"人群 {crowd_id} 计算完成，共 {count} 个用户")
    
    return result_df

def save_to_hbase(result_df, crowd_id):
    """保存结果到HBase"""
    
    # 转换为HBase格式
    hbase_df = result_df.select(
        col("uid").alias("rowkey"),
        lit(f"cf:{crowd_id}").alias("column"), 
        lit("1").alias("value")
    )
    
    # 这里需要使用HBase连接器，简化为打印
    print("保存到HBase...")
    hbase_df.show(10)
    
    return True

if __name__ == "__main__":
    if len(sys.argv) != 4:
        print("用法: spark-submit crowd-compute.py <crowd_id> <rules_json> <dt>")
        sys.exit(1)
    
    crowd_id = sys.argv[1] 
    rules_json = sys.argv[2]
    dt = sys.argv[3]
    
    spark = create_spark_session()
    
    try:
        # 计算人群
        result_df = compute_crowd(spark, crowd_id, rules_json, dt)
        
        # 保存到HBase  
        save_to_hbase(result_df, crowd_id)
        
        print(f"✅ 人群 {crowd_id} 计算任务完成")
        
    except Exception as e:
        print(f"❌ 计算失败: {str(e)}")
        sys.exit(1)
    finally:
        spark.stop()
```

### 2. 人群匹配API

创建 `api/crowd-matching-service.py`:

```python
from flask import Flask, jsonify, request
import happybase
import json
from datetime import datetime

app = Flask(__name__)

# HBase连接配置
HBASE_HOST = 'hbase-master'
HBASE_PORT = 9090

def get_hbase_connection():
    """获取HBase连接"""
    return happybase.Connection(host=HBASE_HOST, port=HBASE_PORT)

@app.route('/api/v1/users/<int:uid>/crowds', methods=['GET'])
def get_user_crowds(uid):
    """查询用户所属人群"""
    try:
        connection = get_hbase_connection()
        table = connection.table('marketing:user_crowd_mapping')
        
        # 查询用户数据
        row = table.row(str(uid).encode())
        
        crowds = []
        for key, value in row.items():
            if key.startswith(b'cf:crowd_'):
                crowd_id = key.decode().replace('cf:', '')
                crowds.append({
                    'crowd_id': crowd_id,
                    'status': value.decode(),
                    'update_time': datetime.now().isoformat()
                })
        
        connection.close()
        
        return jsonify({
            'code': 200,
            'data': {
                'uid': uid,
                'crowds': crowds,
                'total': len(crowds)
            }
        })
        
    except Exception as e:
        return jsonify({
            'code': 500, 
            'message': f'查询失败: {str(e)}'
        }), 500

@app.route('/api/v1/crowds/<crowd_id>/users', methods=['GET'])  
def get_crowd_users(crowd_id):
    """查询人群用户列表"""
    try:
        page = int(request.args.get('page', 1))
        size = int(request.args.get('size', 100))
        
        connection = get_hbase_connection() 
        table = connection.table('marketing:user_crowd_mapping')
        
        # 扫描表格查找包含指定人群的用户
        users = []
        scan_filter = f"SingleColumnValueFilter('cf', 'crowd_{crowd_id}', =, 'binary:1')"
        
        for key, data in table.scan(filter=scan_filter, limit=size):
            uid = key.decode()
            users.append({
                'uid': int(uid),
                'crowd_id': crowd_id,
                'join_time': data.get(b'cf:update_time', b'').decode()
            })
        
        connection.close()
        
        return jsonify({
            'code': 200,
            'data': {
                'crowd_id': crowd_id, 
                'users': users,
                'page': page,
                'size': len(users)
            }
        })
        
    except Exception as e:
        return jsonify({
            'code': 500,
            'message': f'查询失败: {str(e)}'
        }), 500

@app.route('/api/v1/users/batch-query', methods=['POST'])
def batch_query_users():
    """批量查询用户人群"""
    try:
        uids = request.json.get('uids', [])
        
        if not uids or len(uids) > 1000:
            return jsonify({
                'code': 400,
                'message': 'uids参数错误，最多支持1000个用户'
            }), 400
        
        connection = get_hbase_connection()
        table = connection.table('marketing:user_crowd_mapping')
        
        results = []
        for uid in uids:
            row = table.row(str(uid).encode())
            
            crowds = []
            for key, value in row.items():
                if key.startswith(b'cf:crowd_'):
                    crowd_id = key.decode().replace('cf:', '')
                    crowds.append(crowd_id)
            
            results.append({
                'uid': uid,
                'crowds': crowds
            })
        
        connection.close()
        
        return jsonify({
            'code': 200,
            'data': results
        })
        
    except Exception as e:
        return jsonify({
            'code': 500,
            'message': f'批量查询失败: {str(e)}'
        }), 500

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000, debug=True)
```

## 🔄 Airflow DAG示例

创建 `dags/crowd-compute-dag.py`:

```python
from datetime import datetime, timedelta
from airflow import DAG
from airflow.operators.bash_operator import BashOperator
from airflow.operators.python_operator import PythonOperator
import json

default_args = {
    'owner': 'marketing-platform',
    'depends_on_past': False,
    'start_date': datetime(2026, 4, 17),
    'email_on_failure': False,
    'email_on_retry': False,
    'retries': 1,
    'retry_delay': timedelta(minutes=5)
}

def get_crowd_configs():
    """获取需要计算的人群配置"""
    # 这里应该从数据库查询，简化为硬编码
    return [
        {
            'crowd_id': 'crowd_001',
            'rules': {
                'operator': 'AND',
                'conditions': [
                    {'tag': 'age_group', 'operator': 'in', 'values': ['18-25', '26-35']},
                    {'tag': 'consumption_level', 'operator': '>=', 'values': ['高']}
                ]
            }
        }
    ]

dag = DAG(
    'crowd_daily_compute',
    default_args=default_args,
    description='每日人群计算任务',
    schedule_interval='0 2 * * *',  # 每天凌晨2点执行
    catchup=False
)

# 数据质量检查任务
data_quality_check = BashOperator(
    task_id='data_quality_check',
    bash_command='''
    docker exec hive-server beeline -u "jdbc:hive2://hive-server:10000" \
    --silent=true -e "
    USE marketing_platform;
    SELECT COUNT(*) FROM user_tags WHERE dt='{{ ds }}';
    "
    ''',
    dag=dag
)

# 人群计算任务
def create_crowd_compute_task(crowd_config):
    rules_json = json.dumps(crowd_config['rules']).replace('"', '\\"')
    
    return BashOperator(
        task_id=f"compute_{crowd_config['crowd_id']}",
        bash_command=f'''
        docker exec spark-client spark-submit \
        --master yarn \
        --deploy-mode client \
        /opt/spark-jobs/crowd-compute.py \
        "{crowd_config['crowd_id']}" \
        "{rules_json}" \
        "{{{{ ds }}}}"
        ''',
        dag=dag
    )

# 创建人群计算任务
crowd_tasks = []
for crowd_config in get_crowd_configs():
    task = create_crowd_compute_task(crowd_config)
    crowd_tasks.append(task)
    
    # 设置任务依赖
    data_quality_check >> task

# 结果汇总任务
summary_task = BashOperator(
    task_id='compute_summary',
    bash_command='''
    echo "人群计算完成，发送通知..."
    # 这里可以发送钉钉/企微通知
    ''',
    dag=dag
)

# 设置任务依赖
for task in crowd_tasks:
    task >> summary_task
```

## 📞 问题排查

**常见问题解决:**

1. **服务启动失败**: 检查依赖服务是否正常，查看日志定位问题
2. **HDFS异常**: 确认NameNode正常，检查SafeMode状态
3. **HBase连接失败**: 确认ZooKeeper服务正常
4. **Hive查询失败**: 检查MetaStore服务和PostgreSQL连接
5. **内存不足**: 调整Docker资源限制，优化JVM参数

**技术支持:**
- 查看详细架构文档: `大数据圈人营销平台架构文档.md`
- 日志位置: `docker compose logs -f <service-name>`
- 配置文件: `./configs/` 目录下各组件配置
